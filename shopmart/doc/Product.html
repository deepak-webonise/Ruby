def update_stock
      product = FileOperations.search_by_name(self)
      product = product[0].split(';')
      self.id = product[0]
      self.name = product[1]
      self.price = product[2]
      self.company = product[3]
      self.stock = product[4].to_i - 1
      FileOperations.update(self)
end
